// Server-side permission validation
async function validateRolePermission(
  role: UserRole,
  analysisType: AnalysisType,
  targetId: string,
  userId: string
): Promise<boolean> {
  
  // Admin has full access
  if (role === 'admin') return true;
  
  // Founder can only analyze own projects
  if (role === 'founder') {
    const project = await getProject(targetId);
    return project.ownerId === userId;
  }
  
  // VC can analyze projects in deal pipeline
  if (role === 'vc') {
    const hasAccess = await checkDealPipelineAccess(userId, targetId);
    return hasAccess && ['pitch', 'kyb', 'chat', 'financial'].includes(analysisType);
  }
  
  // Exchange can analyze listing applications
  if (role === 'exchange') {
    const hasAccess = await checkListingQueueAccess(userId, targetId);
    return hasAccess && ['kyb', 'pitch', 'financial'].includes(analysisType);
  }
  
  // IDO can analyze launch applications
  if (role === 'ido') {
    const hasAccess = await checkIDOPipelineAccess(userId, targetId);
    return hasAccess && ['pitch', 'kyb', 'financial'].includes(analysisType);
  }
  
  // Agency can analyze client projects
  if (role === 'agency') {
    const hasAccess = await checkClientAccess(userId, targetId);
    return hasAccess && ['pitch', 'kyb', 'chat', 'financial'].includes(analysisType);
  }
  
  // Influencer has verification access (similar to VC but with KYC instead of KYB)
  if (role === 'influencer') {
    return ['pitch', 'kyc', 'chat', 'financial'].includes(analysisType);
  }
  
  return false;
}
