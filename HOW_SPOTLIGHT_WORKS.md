// Service: spotlight-service.ts
static async getActiveSpotlights() {
  // Query for approved spotlights
  const q = query(
    collection(db, 'spotlightApplications'),
    where('status', '==', 'approved'),
    orderBy('startDate', 'desc')
  );
  
  const snapshot = await getDocs(q);
  const now = new Date();
  
  // Filter by date in JavaScript
  const active = snapshot.docs
    .filter(doc => {
      const start = doc.data().startDate.toDate();
      const end = doc.data().endDate.toDate();
      return start <= now && end >= now;
    });
  
  // Separate premium and featured
  const premium = active.find(app => app.slotType === 'premium');
  const featured = active.filter(app => app.slotType === 'featured').slice(0, 3);
  
  return { premium, featured };
}
